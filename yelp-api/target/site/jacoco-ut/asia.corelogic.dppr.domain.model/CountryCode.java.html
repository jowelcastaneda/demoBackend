package asia.corelogic.dppr.domain.model;

import lombok.AllArgsConstructor;
import lombok.Getter;

import java.util.stream.Stream;

@AllArgsConstructor
public enum CountryCode {
    AU("au"),
    NZ("nz");

    @Getter
    private String name;

    public static boolean contains(String test) {
        for (CountryCode code : values()) {
            if (code.name().equalsIgnoreCase(test)) {
                return true;
            }
        }
        return false;
    }

    public static CountryCode of(String name) {
        return Stream.of(CountryCode.values())
                .filter(c -> c.name.equalsIgnoreCase(name))
                .findFirst()
                .orElse(null);
    }
}
