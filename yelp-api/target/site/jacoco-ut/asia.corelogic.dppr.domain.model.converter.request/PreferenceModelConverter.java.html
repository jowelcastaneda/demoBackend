package asia.corelogic.dppr.domain.model.converter.request;


import asia.corelogic.dppr.entity.PreferenceEntity;
import asia.corelogic.dppr.domain.model.converter.DTOConverter;
import asia.corelogic.dppr.domain.model.request.Preference;
import lombok.NoArgsConstructor;
import org.springframework.web.util.HtmlUtils;

@NoArgsConstructor(staticName = "create")
public final class PreferenceModelConverter implements DTOConverter<Preference, PreferenceEntity> {

    @Override
    public PreferenceEntity createFromDto(Preference dto) {
        return PreferenceEntity.builder()
            .footerBackgroundColor(dto.getFooterBackgroundColor())
            .headerBackgroundColor(dto.getHeaderBackgroundColor())
            .footerFontColor(dto.getFooterFontColor())
            .headerFontColor(dto.getHeaderFontColor())
            .bodyBackgroundColor(dto.getBodyBackgroundColor())
            .disclaimer(HtmlUtils.htmlEscape(dto.getDisclaimer()))
            .build();
    }

    public PreferenceEntity createFromDtoWithHeaderLogoUrl(Preference dto, String headerLogoUrl) {
        return createFromDto(dto)
            .toBuilder()
            .headerLogoUrl(headerLogoUrl)
            .build();
    }

}
