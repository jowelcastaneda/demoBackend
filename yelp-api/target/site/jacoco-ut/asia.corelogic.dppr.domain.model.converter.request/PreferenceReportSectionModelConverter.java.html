package asia.corelogic.dppr.domain.model.converter.request;

import asia.corelogic.dppr.entity.PreferenceReportSectionEntity;
import asia.corelogic.dppr.domain.model.converter.GenericModelConverter;
import asia.corelogic.dppr.domain.model.request.Preference;
import asia.corelogic.dppr.domain.model.request.Preference.PreferenceReportSection;
import asia.corelogic.dppr.domain.model.response.PreferenceResponse;
import lombok.NoArgsConstructor;

import java.math.BigInteger;
import java.util.Collection;
import java.util.List;

@NoArgsConstructor(staticName = "create")
public class PreferenceReportSectionModelConverter implements GenericModelConverter<Preference.PreferenceReportSection,
    PreferenceReportSectionEntity, PreferenceResponse.PreferenceReportSection> {

    @Override
    public PreferenceReportSectionEntity createFromDto(PreferenceReportSection dto) {
        return PreferenceReportSectionEntity.builder()
            .reportSectionId(dto.getReportSectionId())
            .enabled(dto.isEnabled())
            .build();
    }

    public List<PreferenceReportSectionEntity> createFromDTOsWithPreferenceId(
        final Collection<? extends PreferenceReportSection> dtos, BigInteger preferenceId) {
        return createFromModifiedDTOs(dtos,
            dto -> createFromDto(dto)
                .toBuilder()
                .preferenceId(preferenceId)
                .build()
        );
    }

    @Override
    public PreferenceResponse.PreferenceReportSection createFromEntity(PreferenceReportSectionEntity entity) {
        return PreferenceResponse.PreferenceReportSection.builder()
            .reportSectionId(entity.getReportSectionId())
            .enabled(entity.getEnabled())
            .build();
    }


}
