package asia.corelogic.dppr.config;

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import springfox.documentation.builders.PathSelectors;
import springfox.documentation.builders.RequestHandlerSelectors;
import springfox.documentation.service.ApiInfo;
import springfox.documentation.service.Contact;
import springfox.documentation.service.Tag;
import springfox.documentation.spi.DocumentationType;
import springfox.documentation.spring.web.plugins.Docket;
import springfox.documentation.swagger2.annotations.EnableSwagger2;

import java.util.ArrayList;

@EnableSwagger2
@Configuration
public class SwaggerConfig {

    public static final String HELLO_WORLD_API = "Hello World Api";
    public static final String PREFERENCE_API = "Preference Api";
    public static final String REPORT_API = "Reports Api";

    @Bean
    public Docket api() {
        return new Docket(DocumentationType.SWAGGER_2)
            .forCodeGeneration(true)
            .select()
            .apis(RequestHandlerSelectors.any())
            .paths(PathSelectors.any())
            .build()
            .apiInfo(metaData())
            .tags(new Tag(HELLO_WORLD_API, "This contains all hello world endpoint"),
                new Tag(PREFERENCE_API, "Preference Api endpoints"),
                new Tag(REPORT_API, "Report Api Services"));
    }

    private ApiInfo metaData() {

        Contact contact = new Contact("Customer Support", "https://corelogic.com.au", "test@gmail.com");

        return new ApiInfo(
            "DPPR API Documentation",
            "",
            "1.0",
            "",
            contact,
            "",
            "",
            new ArrayList<>());

    }

}
