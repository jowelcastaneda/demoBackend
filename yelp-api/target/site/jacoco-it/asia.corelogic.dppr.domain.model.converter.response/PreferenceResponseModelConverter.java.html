package asia.corelogic.dppr.domain.model.converter.response;

import asia.corelogic.dppr.entity.PreferenceEntity;
import asia.corelogic.dppr.domain.model.converter.EntityConverter;
import asia.corelogic.dppr.domain.model.response.PreferenceResponse;
import lombok.NoArgsConstructor;
import org.springframework.web.util.HtmlUtils;

import java.util.List;

import static java.util.Collections.emptyList;

@NoArgsConstructor(staticName = "create")
public final class PreferenceResponseModelConverter implements EntityConverter<PreferenceEntity, PreferenceResponse> {

    @Override
    public PreferenceResponse createFromEntity(PreferenceEntity entity) {
        return PreferenceResponse.builder()
            .preferenceId(entity.getPreferenceId())
            .footerBackgroundColor(entity.getFooterBackgroundColor())
            .headerBackgroundColor(entity.getHeaderBackgroundColor())
            .footerFontColor(entity.getFooterFontColor())
            .headerFontColor(entity.getHeaderFontColor())
            .bodyBackgroundColor(entity.getBodyBackgroundColor())
            .headerLogoUrl(entity.getHeaderLogoUrl())
            .disclaimer(HtmlUtils.htmlUnescape(entity.getDisclaimer()))
            .preferenceReportSections(emptyList())
            .build();
    }

    public PreferenceResponse createFromEntityWithReportSections(
        PreferenceEntity entity, List<PreferenceResponse.PreferenceReportSection> reportSections) {
        return createFromEntity(entity)
            .toBuilder()
            .preferenceReportSections(reportSections)
            .build();
    }
}
