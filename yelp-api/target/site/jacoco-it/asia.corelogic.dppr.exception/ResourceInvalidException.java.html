package asia.corelogic.dppr.exception;

import lombok.Getter;

public final class ResourceInvalidException extends RuntimeException {

    @Getter
    private final String field;

    private ResourceInvalidException(String message, String field) {
        super(message);
        this.field = field;
    }

    public static ResourceInvalidException resourceInvalidField(String field) {
        return new ResourceInvalidException(
            "asia.corelogic.dppr.annotations.validations.EnsureValidField.message",
            field
        );
    }

    public static ResourceInvalidException invalidIntegratorId() {
        return resourceInvalidField("integratorId");
    }

    public static ResourceInvalidException invalidIntegratorUserId() {
        return resourceInvalidField("integratorUserId");
    }

    public static ResourceInvalidException invalidPreferenceId() {
        return resourceInvalidField("preferenceId");
    }

    public static ResourceInvalidException invalidPropertyId() {
        return resourceInvalidField("propertyId");
    }
}
